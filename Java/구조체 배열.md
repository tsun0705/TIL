## 구조체 배열
~~~
Exam[] exams = new Exam[3] // 배열의 공간 생성
exams[0].kor = 30; // 오류 발생

exams[0] =new Exam(); // 객체 참조 선언 후 사용해야 kor, eng, math 사용 가능
exams[1] =new Exam(); 
exams[2] =new Exam(); 
~~~
